grid = np.arange(-np.pi , np.pi, 0.01 )
mat = grid[:,np.newaxis] - x[np.newaxis , :]
x1=Symbol('x') 
equation = 2*x1**2
store = {}
for lambda_ in [0.1, 0.5 , 1] :
    eqn      = Eq( equation , lambda_ )
    value    = np.float32(np.abs(solve( eqn, x1 ))[0])
    weight_k = norm.pdf(mat , loc = 0 , scale = value )
    pred = []
    for grid_idx , weight_idx in zip(grid , weight_k) :
        WLS = LinearRegression()
        WLS.fit(x.reshape(-1,1),
                y.reshape(-1,1), 
                sample_weight= np.squeeze(weight_idx))
        pred.append(float(np.squeeze(WLS.predict(grid_idx.reshape(-1,1)))))
    store[str(lambda_)] = np.array(pred)
